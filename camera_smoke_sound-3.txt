function _init()
	palt(0, false)
	palt(14, true)
	interval=5
	x=30 y=30 f = false sp = 1 leg=33
	sec = 0
	hand=0
	legcheck = false
	smoke=0
	cola={}
	cola_max=100
	cloa_cur=0
	ketchup=false
	flying_ketchup=false
	kfx=false kfy=false
	kx=0 ky=0
	t=0
end

function leg_change()
	if (sec%interval==0) then
		if (leg==33) then 
			leg = 35 legcheck=true
			return
		end
		if (leg==37) then 
			leg = 35 legcheck=false
			return
		end
		if (leg==35) then 
			if (legcheck) then
				leg=37
			else
				leg=33
			end
		end
	end
end

function update_smoke()
	if (sec%interval==0) then
		smoke+=1
		if (smoke==3) then
			smoke=0
		end
	end
end

function throw_ketchup()
	if (sec%2==0) then
		if (kfx==kfy) then
			if (kfx) then
			 kfy=false
			else
				kfy=true
			end
		else
			if (kfx) then
				kfx=false
			else
				kfx=true
			end
		end
	end
	if kf then
		kx-=3
	else
		kx+=3
	end
end

function throw_cola()
	if (sec%2==0) then
		if (kfx==kfy) then
			if (kfx) then
			 kfy=false
			else
				kfy=true
			end
		else
			if (kfx) then
				kfx=false
			else
				kfx=true
			end
		end
	end
	if kf then
		kx-=3
	else
		kx+=3
	end
end

		throw_ketchup()
	end
	if (btn(‚¨ÜÔ∏è)) then 
		y-=1 
		move=true
	end
	if (btn(‚¨áÔ∏è)) then

		if (btn(‚¨ÖÔ∏è)) then
			move=false
			leg=33
			return
		end

		spr(32,kx,ky,1,1,kfx,kfy)
